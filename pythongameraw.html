You will have a double chance of winning if you switch doors.""" if options == prize: remaining = list(set(doors) - set(prize)) door_opener = random.choice(list(set(doors) - set(random.choice(remaining)))) other_option = random.choice(list(set(doors) - set(door_opener) - set(prize))) else: door_opener = random.choice(list(set(doors) - set(options) - set(prize))) other_option = random.choice(list(set(doors) - set(door_opener) - set(options))) print "The door that Monty will open is: %r" % door_opener print """ __________ | __ __ | | | || | | | | || | | | |__||__| | | NO PRIZE | | | || | | | | || | | | | || | | | | || | | | |__||__| | |__________| """ second_chance = raw_input("Would you like to select the third door or keep your initial choice? You have a higher probability of winning if you switch doors. Type 'Yes' or 'No':") if second_chance == "Yes": print "The door you will switch to is: %r" % other_option if other_option == prize: print "Congrats, you win! The prize was behind the other door, %r" % prize if other_option != prize: print "Sorry, the prize was behind the other door, %r" % prize if second_chance == "No": print "You did NOT switch doors." if other_option == prize: print "You lost since you did NOT choose the other option, %r" % prize elif other_option != prize: print "You kept your original choice, you win."
